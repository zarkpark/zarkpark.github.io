  --user '비밀키값이라 자크한테 문의해주세요' \
  --header 'Content-Type: application/json' \
  --header 'Accept: application/json' \
  --data '{
    "payment_data": {
      "capture": "auto",
      "currency": "JPY",
      "intent": "one_time"
    },
    "return_url": "http://localhost:3000/thankyou.html",
    "amount": "1",
    "currency": "JPY",
    "email": "hyunjoon.park@bucketplace.net",
    "payment_types": ["credit_card"]
  }'
